492 $roi_bounds->{$roi}->{$rangecheck}->{max} = $point->{$cut_axes} ; }
493
494 } else {
495 if (!exists $roi_bounds->{$roi}->{min} ||
496 $point->{$cut_axes} < $roi_bounds->{$roi}->{min}) {
497 $roi_bounds->{$roi}->{min} = $point->{$cut_axes} ; }
498
499 if (!exists $roi_bounds->{$roi}->{max} ||
500 $point->{$cut_axes} > $roi_bounds->{$roi}->{max}) {
501 $roi_bounds->{$roi}->{max} = $point->{$cut_axes} ; }
502
503 }
504 }
505 }
506
507# 2. divvy up specified axes into specified number of bins.
508 if (exists $in->{fitted_cuts} && $in->{fitted_cuts} == 1) {
509 foreach my $roi (keys %{$roi_bounds}) {
510 foreach my $rangecheck (keys %{$roi_bounds->{$roi}}) {
511 $roi_bounds->{$roi}->{$rangecheck}->{range} =
512 $roi_bounds->{$roi}->{$rangecheck}->{max} -
513 $roi_bounds->{$roi}->{$rangecheck}->{min} ;
514 if ($roi_bounds->{$roi}->{$rangecheck}->{range} == 0) {
515 $roi_bounds->{$roi}->{$rangecheck}->{range} = 1; }
516 }
517 }
518 } else {
519 foreach my $roi (keys %{$roi_bounds}) {
520 $roi_bounds->{$roi}->{range} = $roi_bounds->{$roi}->{max} -
521 $roi_bounds->{$roi}->{min} ;
522 }
523 }
524
525# 3. iterate over points: assign each point to its bin
526 my $newroidef ;
527 foreach my $xyz (keys %{$roidef->{point2roi}}) {
528 my @t = split(/\,/, $xyz) ;
529 my $point = { x => $t[0], y => $t[1], z => $t[2] } ;
530 my @rangecheck = @t ;
531 $rangecheck[$axes_num->{$cut_axes}] = '*';
532 my $rangecheck = join(',', @rangecheck);
533 foreach my $roi (keys %{$roidef->{point2roi}->{$xyz}}) {
534 my $roibin_no ;
535 if (exists $in->{fitted_cuts} && $in->{fitted_cuts} == 1) {
536 $roibin_no =floor((($point->{$cut_axes} -
537 $roi_bounds->{$roi}->{$rangecheck}->{min})/
538 $roi_bounds->{$roi}->{$rangecheck}->{range})
539 * $in->{numbins}) ;
540 } else {
541 $roibin_no = floor((($point->{$cut_axes} -
542 $roi_bounds->{$roi}->{min}) /
543 $roi_bounds->{$roi}->{range}) * $in->{numbins});
544 }
545
546 if ($roibin_no == $in->{numbins}) {
547 $roibin_no = $roibin_no - 1;}
548 my $newroi = '';
549 if (exists $in->{roi_basename} &&
550 defined $in->{roi_basename}) {
551 $newroi = $in->{roi_basename}."_";
552 }
553 $newroi .= $roi."_part$roibin_no" ;
554# print STDERR "newroi is $newroi e\n" ;
555 $newroidef->{point2roi}->{$xyz}->{$newroi}++ ;
556 }
557 }
558
559# 4. return new point-specified roi specs
560 return $newroidef ;

597sub create_lrcheck_roi {
598# given ROIDEF input, for each ROI, union a flip about sagittal center; split into L/R halves
599# and append new L/R ROIs to the input ->{roidef}
600
601# By default splits all ROIs, optionally uses ->{roi_list}.
602
603# if a -lr-check is given to calc_expr or calc_expr_sim, independent operations for L/R sides are done, and a diff computed.
604
605 my $in = shift ;
606 my $roidef = $in->{roidef} ;
607 my $age = $in->{age} ;
608 my $specs = $in->{specs};
609 if (!exists $in->{specs}) {
610 $specs = alnmnr::getspecs() ;
611 } else {
612 $specs = $in->{specs} ;
613 }
614
615 my $roi_list = [];
616 if (exists $in->{roi_list}) {
617 $roi_list = $in->{roi_list} ;
618 } else {
619 @{$roi_list} = keys %{$roidef->{roi_meta}} ;
620 }
621
622 foreach my $orig_roi (@{$roi_list}) {
623 print STDERR "* creating lrcheck ROI for $orig_roi\n";
624
625# 1. flip across mid-sagittal plane
626 my $flipped_roi = flip_roi({
627 age => $in->{age},
628 roidef => $in->{roidef},
629 roi_list => [$orig_roi],
630 }) ;
631
632 my $new_name = $orig_roi."-LR_joint" ;
633# 2. join with orig ROI
634 my $lr_joint_roi = join_roi({
635 new_name => $new_name,
636 roidef1 => $roidef,
637 roi_name1 => $orig_roi,
638 roidef2 => $flipped_roi,
639 roi_name2 => "flip_".$orig_roi,

643 my $lname = $orig_roi."-LR_part0" ;

645 my $lr_split_roi = midsagittal_split_roi({
646 roidef => $lr_joint_roi,
647 roi_name => $new_name,
648 age => $age,
649 specs => $specs,
650 l_name => $lname,
651 r_name => $rname

654 foreach my $newroi ($lname, $rname) {
655 $roidef->{roi_meta}->{$newroi} = {} ;
656 $roidef->{num_roidef_points}->{$newroi} = 0;

690sub midsagittal_split_roi {
691# splits ->{roi_name} about mid-sagittal plane into ->{l_name} and ->{r_name}
692
693 my $in = shift ;
